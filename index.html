// --- FUNCION CORREGIDA PARA EDITAR O CREAR --- window.nuevaReparacion = async (docId, uf) => { currentUFId = docId; document.getElementById("edit-title").innerText = `Gestionar Reparación UF: ${uf}`; // 1. Limpiamos el formulario por defecto document.querySelectorAll("#modal-edit input, #modal-edit textarea").forEach(i => i.value = ""); document.getElementById("r-estado").value = "Pendiente"; document.getElementById("r-resp").value = "Consorcio"; // 2. Buscamos si ya existe una reparación para esta UF para cargar los datos const q = query(collection(db, "reparaciones"), where("id_uf", "==", docId)); const snap = await getDocs(q); if (!snap.empty) { // Si existe, cargamos los datos del primer registro encontrado const data = snap.docs.data(); document.getElementById("r-desc").value = data.descripcion || ""; document.getElementById("r-ejecutor").value = data.ejecutor || ""; document.getElementById("r-estado").value = data.estado || "Pendiente"; document.getElementById("r-garantia").value = data.garantia || ""; document.getElementById("r-resp").value = data.responsable || "Consorcio"; document.getElementById("r-obs").value = data.observaciones || ""; // Guardamos el ID del documento de reparación para actualizarlo luego window.currentRepId = snap.docs.id; } else { window.currentRepId = null; // Es un registro nuevo } document.getElementById("modal-edit").classList.remove("hidden"); }; // --- GUARDAR O ACTUALIZAR REPARACIÓN --- document.getElementById("btn-save-rep").onclick = async () => { const desc = document.getElementById("r-desc").value; if(!desc) return alert("Falta descripción"); const r = { id_uf: currentUFId, descripcion: desc, ejecutor: document.getElementById("r-ejecutor").value, estado: document.getElementById("r-estado").value, garantia: document.getElementById("r-garantia").value, responsable: document.getElementById("r-resp").value, observaciones: document.getElementById("r-obs").value, fecha: new Date().toLocaleDateString() }; try { if (window.currentRepId) { // Si ya existía, actualizamos el documento existente const { updateDoc, doc } = await import("https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js"); await updateDoc(doc(db, "reparaciones", window.currentRepId), r); } else { // Si no, creamos uno nuevo await addDoc(collection(db, "reparaciones"), r); } cerrarModales(); alert("Datos guardados correctamente"); } catch (e) { console.error("Error al guardar: ", e); } };
